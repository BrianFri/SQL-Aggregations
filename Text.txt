9: GET ANSWER FROM QUERY
SELECT TOTAL,
    CASE
        WHEN TOTAL > 20 THEN "LARGE"
        WHEN TOTAL < 20 AND TOTAL > 5 THEN "MEDIUM"
        ELSE "SMALL"
    END AS "INVOICE SIZE"
FROM Invoice
ORDER BY TOTAL DESC

10: THERE ARE NO LOW/MEDIUM SO WE CHANGED THE NUMBER
SELECT BillingCountry, COUNT(*),
    CASE
        WHEN COUNT(*) > 20 THEN "HIGH"
        WHEN COUNT(*) <= 20 AND COUNT(*) > 10 THEN "MEDIUM"
        ELSE "LOW"
    END AS "PURCHASE FREQUENCY"
FROM Invoice
GROUP BY BillingCountry
ORDER BY COUNT(*) DESC

SELECT CustomerID, Count(*), SUM(Total),
    CASE
        WHEN SUM(Total) < 20 THEN 'Low Value'
        WHEN SUM(Total) BETWEEN 20 and 40 THEN 'Medium Value'
        ELSE 'High Value'
    END AS CustomerStatus
FROM Invoice
GROUP BY CustomerId

13.
SELECT COUNT(*),
    CASE
        WHEN Total < 2 THEN 'Under $2'
        WHEN Total BETWEEN 2 and 5 THEN '$2-$5'
        WHEN Total BETWEEN 5 and 10 THEN '$5-$10'
        ELSE 'Over $10'
    END AS Invoice_Value_Distribution
FROM Invoice
GROUP BY Invoice_Value_Distribution
