9: GET ANSWER FROM QUERY
SELECT TOTAL,
    CASE
        WHEN TOTAL > 20 THEN "LARGE"
        WHEN TOTAL < 20 AND TOTAL > 5 THEN "MEDIUM"
        ELSE "SMALL"
    END AS "INVOICE SIZE"
FROM Invoice
ORDER BY TOTAL DESC

10: THERE ARE NO LOW/MEDIUM SO WE CHANGED THE NUMBER
SELECT BillingCountry, COUNT(*),
    CASE
        WHEN COUNT(*) > 20 THEN "HIGH"
        WHEN COUNT(*) <= 20 AND COUNT(*) > 10 THEN "MEDIUM"
        ELSE "LOW"
    END AS "PURCHASE FREQUENCY"
FROM Invoice
GROUP BY BillingCountry
ORDER BY COUNT(*) DESC
